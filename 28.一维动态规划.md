
[P9242 [蓝桥杯 2023 省 B] 接龙数列 - 洛谷](https://www.luogu.com.cn/problem/P9242)

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,dp[10],maxn;
string a;//为了方便取头尾，可以以字符串形式存储
signed main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a;
		int ln=a.length();
		dp[a[ln-1]-'0']=max(dp[a[ln-1]-'0'],dp[a[0]-'0']+1);
	}
	for(int i=0;i<=9;i++) maxn=max(maxn,dp[i]);
	cout<<n-maxn;
	return 0;
}
```

[U549625 小苯的区间和疑惑 - 洛谷](https://www.luogu.com.cn/problem/U549625#ide)
```cpp
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

long long arr[200005];
long long L[200005]; // 以 i 结尾的最大子段和
long long R[200005]; // 以 i 开始的最大子段和

void solve() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> arr[i];

    // 从左往右跑 Kadane
    // L[i] = max(arr[i], L[i-1] + arr[i])
    L[0] = -2e18; // 极小值初始化
    long long current_max = 0;
    for (int i = 1; i <= n; i++) {
        current_max = max(arr[i], current_max + arr[i]);
        L[i] = current_max;
    }

    // 从右往左跑 Kadane
    // R[i] = max(arr[i], R[i+1] + arr[i])
    R[n + 1] = -2e18;
    current_max = 0;
    for (int i = n; i >= 1; i--) {
        current_max = max(arr[i], current_max + arr[i]);
        R[i] = current_max;
    }

    // 输出经过 i 的最大和
    for (int i = 1; i <= n; i++) {
        cout << (L[i] + R[i] - arr[i]) << (i == n ? "" : " ");
    }
    cout << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    return 0;
}
```

[P1115 最大子段和 - 洛谷](https://www.luogu.com.cn/problem/solution/P1115)
```cpp
#include<bits/stdc++.h>
using namespace std;
int n,a,b,i,ans=-2147483647;
int main(){
   cin>>n;
   for(i=1;i<=n;i++){
   	cin>>a;
   	if(i==1) b=a;
   	else b=max(a,a+b);
   	ans=max(ans,b);
   }
   cout<<ans;
   return 0;
}
```
