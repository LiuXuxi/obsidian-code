class Solution {

public:

    int numIslands(vector<vector<char>>& grid) {

        int ans = 0;

        for (int i = 0; i < grid.size(); i++) {

            for (int j = 0; j < grid[0].size(); j++) {

                if(grid[i][j]=='1'){

                    ++ans;

                    dfs(grid, i, j);

                }

            }

        }

        return ans;

    }

    void dfs(vector<vector<char>>& grid, int i, int j) {

        if (i < 0 || i >= grid.size() || j < 0 || j >= grid[0].size() || grid[i][j] != '1')return;

        grid[i][j]='0';

        dfs(grid, i - 1, j);

        dfs(grid, i + 1, j);

        dfs(grid, i, j - 1);

        dfs(grid, i, j + 1);

    }

};

#include<iostream>
using namespace std;

char grid[105][105],ans[105][105];
int n;
string s{"yizhong"};
int dx[]={-1,-1,-1,0,0,1,1,1};
int dy[]={-1,0,1,-1,1,-1,0,1};
bool check;

void dfs(int x,int y,int cnt,int dir){
    if(cnt==7){
        ans[x][y]=grid[x][y];
        check=true;
        return;
    }
    int nx=x+dx[dir];
    int ny=y+dy[dir];
    if(nx<0||nx>=n||ny<0||ny>=n)return;
    if(grid[nx][ny]!=s[cnt])return;
    dfs(nx,ny,cnt+1,dir);
    if(check)ans[x][y]=grid[x][y];
}

int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cin>>grid[i][j];
            ans[i][j]='*';
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(grid[i][j]=='y'){
                for(int dir=0;dir<8;dir++){
                    check=0;
                    dfs(i,j,1,dir);
                    if(check)ans[i][j]=grid[i][j];
                }
            }
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cout<<ans[i][j];
        }
        cout<<endl;
    }
    return 0;
}

#include<iostream>
using namespace std;

int grid[35][35];
int n;

void dfs(int i,int j){
    if(i<0||i>=n||j<0||j>=n||grid[i][j]!=0)return;
    grid[i][j]=3;
    dfs(i-1,j);
    dfs(i+1,j);
    dfs(i,j-1);
    dfs(i,j+1);
}

int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cin>>grid[i][j];
        }
    }
    for(int i=0;i<n;i++){
        dfs(i,0);
        dfs(i,n-1);
        dfs(0,i);
        dfs(n-1,i); 
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(grid[i][j]==3)cout<<0<<" ";
            else if(grid[i][j]==0)cout<<2<<" ";
            else cout<<grid[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
