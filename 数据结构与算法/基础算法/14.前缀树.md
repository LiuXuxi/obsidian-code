[208. 实现 Trie (前缀树) - 力扣（LeetCode）](https://leetcode.cn/problems/implement-trie-prefix-tree/)

```cpp
class Trie {

public:

    Trie() {

        root = new Node;

    }

  

    void insert(string word) {

        root->pass++;

        Node* node = root;

        for (auto& c : word) {

            int path = c - 'a';

            if (node->nexts[path] == nullptr) {

                node->nexts[path] = new Node;

            }

            node = node->nexts[path];

            ++node->pass;

        }

        ++node->end;

    }

  

    bool search(string word) {

        Node* node = root;

        for (auto& c : word) {

            int path = c - 'a';

            if (node->nexts[path] == nullptr)return false;

            node = node->nexts[path];

        }

        return node->end != 0  ;

    }

  

    bool startsWith(string prefix) {

        Node* node = root;

        for (auto& c : prefix) {

            int path = c - 'a';

            if (node->nexts[path] == nullptr)return false;

            node = node->nexts[path];

        }

        return true;

    }

  

private:

    class Node {

    public:

        int pass;

        int end;

        unordered_map<int, Node*>nexts;

  

    public:

        Node():pass(0),end(0),nexts(unordered_map<int,Node*>()) {}

    };

  

public:

    Node* root;

};
```
