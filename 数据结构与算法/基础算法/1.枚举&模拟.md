## 1.1枚举——基于已有知识来猜测答案的一种求解策略
- 不断地猜测，从可能的答案集合中一个一个尝试，然后再判断题目的条件是否成立，枚举每一种可能性
1. 找对答案集合
2. 答案成立条件
3. 减小枚举范围
4. 选择合适的枚举顺序
## 1.2模拟——直接模拟题目要求中的操作来求解
- 模拟题目通常码量大，操作多，难以查错
## 1.3暴力——枚举和模拟的统称，可以不做严格区分
- 暴力搜索（暴搜）：用搜索算法枚举答案（找到符合条件的答案）
- 三大玄学解题方法：暴力求解法，直观目测法，心灵感应法
## 1.4做题步骤
1. 读题意，分析题意
2. 看示例（样例），分析样例
3. 看数据范围，确定时间复杂度（1s=10^8)数据范围
4. 找思路，写代码，本地运行测试，提交

## 2.1枚举右，维护左

[624. 数组列表中的最大距离 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-distance-in-arrays/)

> 枚举不止数值，还可以是**数组**。

```cpp
class Solution {

public:

    int maxDistance(vector<vector<int>>& arrays) {

        int ans=0;

        int minV=INT_MAX/2,maxV=INT_MIN/2;

        for(auto&array:arrays){

            ans=max({ans,array.back()-minV,maxV-array[0]});

            maxV=max(maxV,array.back());

            minV=min(minV,array[0]);

        }

        return ans;

    }

};
```


[1128. 等价多米诺骨牌对的数量 - 力扣（LeetCode）](https://leetcode.cn/problems/number-of-equivalent-domino-pairs/submissions/705395356/)

> **关键点：** 二元组的每个值不超过10。
> 哈希表的键值不能为数组，该如何表示键值？
> 将数组表示为数值，例如二元组的两个数拼凑起来，这样同样符合（满足相同条件的值所对应的键值相同的）键值的条件。

```cpp
class Solution {

public:

    int numEquivDominoPairs(vector<vector<int>>& dominoes) {

        vector<int>cnt(100,0);

        int ans=0;

        for(auto&dominoy:dominoes){

            int val=dominoy[0]<=dominoy[1]?dominoy[0]*10+dominoy[1]:dominoy[1]*10+dominoy[0];

            ans+=cnt[val];

            cnt[val]++;

        }

        return ans;

    }

};
```

[2001. 可互换矩形的组数 - 力扣（LeetCode）](https://leetcode.cn/problems/number-of-pairs-of-interchangeable-rectangles/)
> 哈希表的键值不能为数组，该如何表示键值？
> 唯一性为`height/weight`，但注意：用`double`存储和计算键值，而不是`int`。
```cpp
class Solution {

public:

    long long interchangeableRectangles(vector<vector<int>>& rectangles) {

        unordered_map<double,int>cnt;

        long long ans=0;

        for(auto&rectangle:rectangles){

            double value=(double)rectangle[1]/rectangle[0];

            ans+=cnt[value];

            cnt[value]++;

        }

        return ans;

    }

};
```

[3623. 统计梯形的数目 I - 力扣（LeetCode）](https://leetcode.cn/problems/count-number-of-trapezoids-i/submissions/705776594/)

> 枚举不止是`nums`，有可能是哈希表。

```cpp
class Solution {

public:

    int countTrapezoids(vector<vector<int>>& points) {

        const int mod=1e9+7;

        unordered_map<int,int>cnt;

        for(auto&point:points)cnt[point[1]]++;

        long long ans=0,sum=0;

        for(auto&[_,y]:cnt){

            long long lines=(long long)y*(y-1)/2;

            ans+=((lines%mod)*(sum%mod))%mod;

            sum+=lines;

        }

        return ans%mod;

    }

};
```


[3584. 子序列首尾元素的最大乘积 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-product-of-first-and-last-elements-of-a-subsequence/)

> 脑筋急转弯
```cpp
class Solution {

public:

    long long maximumProduct(vector<int>& nums, int m) {

        long long ans = LLONG_MIN;

        int mn = INT_MAX, mx = INT_MIN;

        for (int i = m - 1; i < nums.size(); i++) {

            // 维护左边 [0,i-m+1] 中的最小值和最大值

            int y = nums[i - m + 1];

            mn = min(mn, y);

            mx = max(mx, y);

            // 枚举右

            long long x = nums[i];

            ans = max({ans, x * mn, x * mx});

        }

        return ans;

    }

};
```

[3371. 识别数组中的最大异常值 - 力扣（LeetCode）](https://leetcode.cn/problems/identify-the-largest-outlier-in-an-array/submissions/706038509/)
> 找出数学关系，枚举异常值或元素和。
```cpp
class Solution {

public:

    int getLargestOutlier(vector<int>& nums) {

        int sum=0;

        unordered_map<int,int>cnt;

        for(auto&num:nums){

            sum+=num;

            cnt[num]++;

        }

        int ans=INT_MIN;

        for(auto&num:nums){

            --cnt[num];

            if((sum-num)%2==0&&cnt[(sum-num)/2]>0)ans=max(ans,num);

            cnt[num]++;

        }

        return ans;

    }

};
```

## 2.2枚举中间

> 枚举中间+前后缀分解

[2909. 元素和最小的山形三元组 II - 力扣（LeetCode）](https://leetcode.cn/problems/minimum-sum-of-mountain-triplets-ii/)

```cpp
class Solution {

public:

    int minimumSum(vector<int>& nums) {

        int n=nums.size();

        vector<int>suf(n);

        suf[n-1]=nums[n-1];

        for(int i=n-2;i>=0;i--)suf[i]=min(suf[i+1],nums[i]);

        int ans=__INT_MAX__;

        for(int i=0,pre=nums[0];i<n-1;i++){

            if(pre<nums[i]&&suf[i+1]<nums[i])ans=min(ans,pre+nums[i]+suf[i+1]);

            pre=min(pre,nums[i]);

        }

        return ans==__INT_MAX__?-1:ans;

    }

};
```

[1930. 长度为 3 的不同回文子序列 - 力扣（LeetCode）](https://leetcode.cn/problems/unique-length-3-palindromic-subsequences/submissions/706239734/)

**枚举两侧：** 

```cpp
class Solution {

public:

    int countPalindromicSubsequence(string s) {

        int ans=0;

        for(char c='a';c<='z';c++){

            int pre=s.find(c);

            if(pre==s.npos)continue;

            int suf=s.rfind(c);

            vector<bool>visited(26,false);

            for(int i=pre+1;i<suf;i++){

                if(!visited[s[i]-'a']){

                    visited[s[i]-'a']=true;

                    ++ans;

                }

            }

        }

        return ans;

    }

};
```


**枚举中间：** 

```cpp
class Solution {

public:

    int countPalindromicSubsequence(string s) {

        vector<int>suf(26,0);

        int n=s.size();

        for(int i=1;i<n;i++)suf[s[i]-'a']++;

        vector<int>pre(26,0);

        vector<vector<bool>>has(26,vector<bool>(26,false));

        int ans=0;

        for(int i=1;i<n-1;i++){

            pre[s[i-1]-'a']++;

            suf[s[i]-'a']--;

            for(char c='a';c<='z';c++){

                if(pre[c-'a']&&suf[c-'a']&&!has[s[i]-'a'][c-'a']){

                    has[s[i]-'a'][c-'a']=true;

                    ++ans;

                }

            }

        }

        return ans;

    }

};
```
