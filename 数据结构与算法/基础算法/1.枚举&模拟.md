## 1.1枚举——基于已有知识来猜测答案的一种求解策略
- 不断地猜测，从可能的答案集合中一个一个尝试，然后再判断题目的条件是否成立，枚举每一种可能性
1. 找对答案集合

3. 减小枚举范围
4. 选择合适的枚举顺序
## 1.2模拟——直接模拟题目要求中的操作来求解
- 模拟题目通常码量大，操作多，难以查错
## 1.3暴力——枚举和模拟的统称，可以不做严格区分
- 暴力搜索（暴搜）：用搜索算法枚举答案（找到符合条件的答案）
- 三大玄学解题方法：暴力求解法，直观目测法，心灵感应法
## 1.4做题步骤
1. 读题意，分析题意

3. 看数据范围，确定时间复杂度（1s=10^8)数据范围

[624. 数组列表中的最大距离 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-distance-in-arrays/)

> 枚举不止数值，还可以是**数组**。

```cpp
class Solution {

public:

    int maxDistance(vector<vector<int>>& arrays) {

        int ans=0;

        int minV=INT_MAX/2,maxV=INT_MIN/2;

        for(auto&array:arrays){

            ans=max({ans,array.back()-minV,maxV-array[0]});

            maxV=max(maxV,array.back());

            minV=min(minV,array[0]);

        }

        return ans;

    }

};
```


[1128. 等价多米诺骨牌对的数量 - 力扣（LeetCode）](https://leetcode.cn/problems/number-of-equivalent-domino-pairs/submissions/705395356/)

> **关键点：** 二元组的每个值不超过10。
> 哈希表的键值不能为数组，该如何表示键值？
> 将数组表示为数值，例如二元组的两个数拼凑起来，这样同样符合（满足相同条件的值所对应的键值相同的）键值的条件。

```cpp
class Solution {

public:

    int numEquivDominoPairs(vector<vector<int>>& dominoes) {

        vector<int>cnt(100,0);

        int ans=0;

        for(auto&dominoy:dominoes){

            int val=dominoy[0]<=dominoy[1]?dominoy[0]*10+dominoy[1]:dominoy[1]*10+dominoy[0];

            ans+=cnt[val];

            cnt[val]++;

        }

        return ans;

    }

};
```

[3623. 统计梯形的数目 I - 力扣（LeetCode）](https://leetcode.cn/problems/count-number-of-trapezoids-i/submissions/705776594/)

> 枚举不止是`nums`，有可能是哈希表。

```cpp
class Solution {

public:

    int countTrapezoids(vector<vector<int>>& points) {

        const int mod=1e9+7;

        unordered_map<int,int>cnt;

        for(auto&point:points)cnt[point[1]]++;

        long long ans=0,sum=0;

        for(auto&[_,y]:cnt){

            long long lines=(long long)y*(y-1)/2;

            ans+=((lines%mod)*(sum%mod))%mod;

            sum+=lines;

        }

        return ans%mod;

    }

};
```


[3584. 子序列首尾元素的最大乘积 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-product-of-first-and-last-elements-of-a-subsequence/)

> 脑筋急转弯
```cpp
class Solution {

public:

    long long maximumProduct(vector<int>& nums, int m) {

        long long ans = LLONG_MIN;

        int mn = INT_MAX, mx = INT_MIN;

        for (int i = m - 1; i < nums.size(); i++) {

            // 维护左边 [0,i-m+1] 中的最小值和最大值

            int y = nums[i - m + 1];

            mn = min(mn, y);

            mx = max(mx, y);

            // 枚举右

            long long x = nums[i];

            ans = max({ans, x * mn, x * mx});

        }

        return ans;

    }

};
```

[3371. 识别数组中的最大异常值 - 力扣（LeetCode）](https://leetcode.cn/problems/identify-the-largest-outlier-in-an-array/submissions/706038509/)
> 找出数学关系，枚举异常值或元素和。
```cpp
class Solution {

public:

    int getLargestOutlier(vector<int>& nums) {

        int sum=0;

        unordered_map<int,int>cnt;

        for(auto&num:nums){

            sum+=num;

            cnt[num]++;

        }

        int ans=INT_MIN;

        for(auto&num:nums){

            --cnt[num];

            if((sum-num)%2==0&&cnt[(sum-num)/2]>0)ans=max(ans,num);

            cnt[num]++;

        }

        return ans;

    }

};
```

**总结：** 向右枚举的同时，维护左边的数据或者说处理左边的数据。

那么重点就在于两点：**向右枚举的策略**以及**维护左边的策略** 。

**向右枚举的策略：** 
1. ==先仔细思考枚举的对象。==是一维数组中的元素，还是二维数组中一维数组的特定元素，还是自己创建的哈希表等等。
2. ==再琢磨琢磨枚举的范围。==是0->n-1，还是1->n-1，还是1->n-2，这也取决于你的维护策略。比如只维护左边，可能是0->n-1或1->n-1，但如果还需维护右边，则可能为1->n-2，具体看题目要求。

**维护左边的策略：** 事实上有时不仅仅是只维护左边，可能同时维护左边和右边数据，这往往与前后缀分解结合。
1. ==思考维护数据的容器是什么。==如果只是想维护最大最小值，可能只需要一个变量；如果想维护符合条件的数量，可能需要一个数组，或者一个哈希表。
2. ==如果维护哈希表，键值是什么？==我们需要思考数据的统一特性（共同且唯一的特性），比如同一个数值出现的次数，以数值本身为键值；==而当找到的键值不符合哈希表键值要求是该如何处理？==比如为pair对时，考虑二元组进行压缩；为分数时，用小数double替代，等等。

[2909. 元素和最小的山形三元组 II - 力扣（LeetCode）](https://leetcode.cn/problems/minimum-sum-of-mountain-triplets-ii/)

```cpp
class Solution {

public:

    int minimumSum(vector<int>& nums) {

        int n=nums.size();

        vector<int>suf(n);

        suf[n-1]=nums[n-1];

        for(int i=n-2;i>=0;i--)suf[i]=min(suf[i+1],nums[i]);

        int ans=__INT_MAX__;

        for(int i=0,pre=nums[0];i<n-1;i++){

            if(pre<nums[i]&&suf[i+1]<nums[i])ans=min(ans,pre+nums[i]+suf[i+1]);

            pre=min(pre,nums[i]);

        }

        return ans==__INT_MAX__?-1:ans;

    }

};
```

[456. 132 模式 - 力扣（LeetCode）](https://leetcode.cn/problems/132-pattern/)

> 枚举i还是j还是K？
> 维护suf不只是哈希表，也可以是有序表
```cpp
class Solution {

public:

    bool find132pattern(vector<int>& nums) {

        int n=nums.size();

        if(n<3)return false;

        multiset<int>suf;

        for(int i=2;i<n;i++)suf.insert(nums[i]);

        for(int i=1,pre=nums[0];i<n-1;i++){

            if(nums[i]>pre){

                //使用multiset关联式容器自带的二分算法，而不是<algorithm>的二分

                auto it=suf.upper_bound(pre);

                if(it!=suf.end()&&*it<nums[i])return true;

            }

            pre=min(pre,nums[i]);

            //为什么要加find()?

            //find会返回一个指向nums[j+1]的迭代器，指定删除一个元素

            //如果不加find，只是erase(nums[j+1])，会删除所有元素等于nums[j+1]的元素

            suf.erase(suf.find(nums[i+1]));

        }

        return false;

    }

};
```

[3067. 在带权树网络中统计可连接服务器对数目 - 力扣（LeetCode）](https://leetcode.cn/problems/count-pairs-of-connectable-servers-in-a-weighted-tree-network/)

```cpp
class Solution {

public:

    vector<int> countPairsOfConnectableServers(vector<vector<int>>& edges, int signalSpeed) {

        int n=edges.size()+1;

        vector<vector<pair<int,int>>>grid(n);

        for(auto&edge:edges){

            int x=edge[0],y=edge[1],w=edge[2];

            grid[x].push_back({y,w});

            grid[y].push_back({x,w});

        }

  

        auto dfs = [&](this auto&& dfs,int i,int f,int sum) ->int {

            int cnt=sum%signalSpeed?0:1;

            for(auto&[y,w]:grid[i]){

                if(y!=f)cnt+=dfs(y,i,w+sum);

            }

            return cnt;

        };

  

        vector<int>ans(n);

        for(int i=0;i<n;i++){

            if(grid[i].size()==1)continue;

            int sum=0;

            for(auto&[y,w]:grid[i]){

                int cnt=dfs(y,i,w);

                ans[i]+=cnt*sum;

                sum+=cnt;

            }

        }

        return ans;

    }

  

};
```

```cpp
class Solution {

public:

    vector<vector<int>> sortMatrix(vector<vector<int>>& grid) {

        int m=grid.size(),n=grid[0].size();

        for(int k=1;k<m+n;k++){

            int min_j=max(0,n-k);

            int max_j=min(m+n-k-1,n-1);

            vector<int>arr;

            for(int j=min_j;j<=max_j;j++)arr.push_back(grid[k+j-n][j]);

            if(min_j>0)sort(arr.begin(),arr.end());

            else sort(arr.begin(),arr.end(),[](const int a,const int b){return a>b;});

            for(int j=min_j;j<=max_j;j++)grid[k+j-n][j]=arr[j-min_j];

        }

        return grid;

    }

};
```

```cpp
class Solution {

public:

    vector<vector<int>> differenceOfDistinctValues(vector<vector<int>>& grid) {

        int m=grid.size(),n=grid[0].size();

        vector<vector<int>>ans(m,vector<int>(n));

        unordered_set<int>st;

        //k=i-j+n;

        for(int k=1;k<m+n;k++){

            //j=i+n-k;

            int min_j=max(0,n-k);

            int max_j=min(n-1,m+n-k-1);

            st.clear();

            for(int j=min_j;j<=max_j;j++){

                //topleft

                ans[k+j-n][j]=st.size();

                st.insert(grid[k+j-n][j]);

            }

            st.clear();

            for(int j=max_j;j>=min_j;j--){

                //bottomright

                ans[k+j-n][j]=abs(ans[k+j-n][j]-(int)st.size());

                st.insert(grid[k+j-n][j]);

            }

        }

        return ans;

    }

};
```
