
# 1.经典用法

[84. 柱状图中最大的矩形 - 力扣（LeetCode）](https://leetcode.cn/problems/largest-rectangle-in-histogram/submissions/696368142/)

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int>s;
        int n = heights.size();
        int left = 0, ans = 0;
        for (int i = 0; i < heights.size(); i++) {
            while (!s.empty() && heights[s.top()] >= heights[i]) {
                int cur = s.top();
                s.pop();
                left = s.empty() ? -1 : s.top();
                ans = max(ans, heights[cur] * (i - left - 1));
            }
            s.push(i);
        }
        while (!s.empty()) {
            int cur = s.top();
            s.pop();
            left = s.empty() ? -1 : s.top();
            ans = max(ans, heights[cur] * (n - left - 1));
        }
        return ans;
    }
};
```

```cpp
class Solution {

public:

    int maxWidthRamp(vector<int>& nums) {

        stack<int>s;

        s.push(0);

        int ans=0;

        for (int i = 1; i < nums.size(); i++) {

            if (nums[i] < nums[s.top()])s.push(i);

        }

        for(int i=nums.size()-1;i>=0;--i){

            while(!s.empty()&&nums[s.top()]<=nums[i]){

                ans=max(ans,i-s.top());

                s.pop();

            }

        }

        return ans;

    }

};
```

[316. 去除重复字母 - 力扣（LeetCode）](https://leetcode.cn/problems/remove-duplicate-letters/)

```cpp
class Solution {

public:

    string removeDuplicateLetters(string s) {

        stack<char>stk;

        vector<bool>enrolled(26);

        vector<int>cnt(26);

        for(auto&c:s)cnt[c-'a']++;

        for(auto&c:s){

            if(!enrolled[c-'a']){

                while(!stk.empty()&&stk.top()>c&&cnt[stk.top()-'a']>0){

                    enrolled[stk.top()-'a']=false;

                    stk.pop();

                }

                enrolled[c-'a']=true;

                stk.push(c);

            }

            --cnt[c-'a'];

        }

        string ans;

        while(!stk.empty()){

            ans=stk.top()+ans;

            stk.pop();

        }

        return ans;

    }

};
```

[大雨吃小鱼_牛客题霸_牛客网](https://www.nowcoder.com/practice/77199defc4b74b24b8ebf6244e1793de)

```cpp
#include<iostream>

#include<stack>

#include<vector>

using namespace std;

  

int solve(vector<int>& nums) {

    int ans = 0;

    stack<pair<int,int>>s;

    for (int i = nums.size() - 1; i >= 0; i--) {

        int curturn = 0;

        //出栈阶段

        while (!s.empty() && s.top().first < nums[i]) {

            curturn = max(curturn + 1, s.top().second);

            s.pop();

        }

        //入栈阶段

        s.push({ nums[i],curturn });

        ans = max(ans, curturn);

    }

    return ans;

}

  

int main() {

    int n;

    cin >> n;

    vector<int>nums(n);

    for (auto& num : nums)cin >> num;

    cout << solve(nums);

    return 0;

}
```
