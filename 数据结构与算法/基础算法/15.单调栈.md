
[84. 柱状图中最大的矩形 - 力扣（LeetCode）](https://leetcode.cn/problems/largest-rectangle-in-histogram/submissions/696368142/)

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int>s;
        int n = heights.size();
        int left = 0, ans = 0;
        for (int i = 0; i < heights.size(); i++) {
            while (!s.empty() && heights[s.top()] >= heights[i]) {
                int cur = s.top();
                s.pop();
                left = s.empty() ? -1 : s.top();
                ans = max(ans, heights[cur] * (i - left - 1));
            }
            s.push(i);
        }
        while (!s.empty()) {
            int cur = s.top();
            s.pop();
            left = s.empty() ? -1 : s.top();
            ans = max(ans, heights[cur] * (n - left - 1));
        }
        return ans;
    }
};
```
