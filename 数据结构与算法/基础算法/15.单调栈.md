[84. 柱状图中最大的矩形 - 力扣（LeetCode）](https://leetcode.cn/problems/largest-rectangle-in-histogram/submissions/696368142/)

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int>s;
        int n = heights.size();
        int left = 0, ans = 0;
        for (int i = 0; i < heights.size(); i++) {
            while (!s.empty() && heights[s.top()] >= heights[i]) {
                int cur = s.top();
                s.pop();
                left = s.empty() ? -1 : s.top();
                ans = max(ans, heights[cur] * (i - left - 1));
            }
            s.push(i);
        }
        while (!s.empty()) {
            int cur = s.top();
            s.pop();
            left = s.empty() ? -1 : s.top();
            ans = max(ans, heights[cur] * (n - left - 1));
        }
        return ans;
    }
};
```

[1381. 设计一个支持增量操作的栈 - 力扣（LeetCode）](https://leetcode.cn/problems/design-a-stack-with-increment-operation/solutions/241895/she-ji-yi-ge-zhi-chi-zeng-liang-cao-zuo-de-zhan-by/)
> 类似于栈+前缀和（辅助栈）
```cpp
class CustomStack {
public:
    vector<int> stk, add;
    int top;

    CustomStack(int maxSize) {
        stk.resize(maxSize);
        add.resize(maxSize);
        top = -1;
    }
    
    void push(int x) {
        if (top != stk.size() - 1) {
            ++top;
            stk[top] = x;
        }
    }
    
    int pop() {
        if (top == -1) {
            return -1;
        }
        int ret = stk[top] + add[top];
        if (top != 0) {
            add[top - 1] += add[top];
        }
        add[top] = 0;
        --top;
        return ret;
    }
    
    void increment(int k, int val) {
        int lim = min(k - 1, top);
        if (lim >= 0) {
            add[lim] += val;
        }
    }
};
```

[F-小苯的好段计数_“⌬杯”蓝桥杯大赛省赛模拟赛](https://ac.nowcoder.com/acm/contest/132328/F)

> 维护左右两边最近的最值

```cpp
#include <iostream>

#include <vector>

#include <stack>

#include <algorithm>

  

using namespace std;

  

void solve() {

    int n;

    cin >> n;

    vector<int> a(n + 1);

    for (int i = 1; i <= n; i++) cin >> a[i];

  

    vector<int> L(n + 1, 0), R(n + 1, n + 1);

    stack<int> s;

  

    // 单调栈求左边第一个大于 a[i] 的位置

    for (int i = 1; i <= n; i++) {

        while (!s.empty() && a[s.top()] < a[i]) s.pop();

        if (!s.empty()) L[i] = s.top();

        s.push(i);

    }

  

    while (!s.empty()) s.pop();

  

    // 单调栈求右边第一个大于 a[i] 的位置

    for (int i = n; i >= 1; i--) {

        while (!s.empty() && a[s.top()] < a[i]) s.pop();

        if (!s.empty()) R[i] = s.top();

        s.push(i);

    }

  

    long long ans = 0;

    for (int i = 1; i <= n; i++) {

        if(R[i]-L[i]-1>=a[i])++ans;

    }

    cout << ans << endl;

}

  

int main() {

    ios::sync_with_stdio(false);

    cin.tie(nullptr);

    int t;

    cin >> t;

    while (t--) solve();

    return 0;

}
```

[316. 去除重复字母 - 力扣（LeetCode）](https://leetcode.cn/problems/remove-duplicate-letters/)

```cpp
class Solution {

public:

    string removeDuplicateLetters(string s) {

        stack<char>stk;

        vector<bool>enrolled(26);

        vector<int>cnt(26);

        for(auto&c:s)cnt[c-'a']++;

        for(auto&c:s){

            if(!enrolled[c-'a']){

                while(!stk.empty()&&stk.top()>c&&cnt[stk.top()-'a']>0){

                    enrolled[stk.top()-'a']=false;

                    stk.pop();

                }

                enrolled[c-'a']=true;

                stk.push(c);

            }

            --cnt[c-'a'];

        }

        string ans;

        while(!stk.empty()){

            ans=stk.top()+ans;

            stk.pop();

        }

        return ans;

    }

};
```

```cpp
#include<iostream>

#include<stack>

#include<vector>

using namespace std;

  

int solve(vector<int>& nums) {

    int ans = 0;

    stack<pair<int,int>>s;

    for (int i = nums.size() - 1; i >= 0; i--) {

        int curturn = 0;

        //出栈阶段

        while (!s.empty() && s.top().first < nums[i]) {

            curturn = max(curturn + 1, s.top().second);

            s.pop();

        }

        //入栈阶段

        s.push({ nums[i],curturn });

        ans = max(ans, curturn);

    }

    return ans;

}

  

int main() {

    int n;

    cin >> n;

    vector<int>nums(n);

    for (auto& num : nums)cin >> num;

    cout << solve(nums);

    return 0;

}
```

[1504. 统计全 1 子矩形 - 力扣（LeetCode）](https://leetcode.cn/problems/count-submatrices-with-all-ones/description/)

```cpp
class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        vector<int>heights(mat[0].size());
        int ans = 0;
        for (int i = 0; i < mat.size(); i++) {
            for (int j = 0; j < mat[i].size(); j++) {
                heights[j] = mat[i][j] == 1 ? heights[j] + 1 : 0;
            }
            ans += f(heights);
        }
        return ans;
    }

    int f(vector<int>& heights) {
        int ans = 0;
        stack<int>s;
        for (int i = 0; i < heights.size(); i++) {
            while (!s.empty() && heights[s.top()] >= heights[i]) {
                int cur = s.top();
                s.pop();
                if (heights[cur] > heights[i]) {
                    int left = s.empty() ? -1 : s.top();
                    int bottom = max(heights[i], s.empty() ? 0 : heights[left]);
                    ans += (heights[cur] - bottom) * (i - left - 1) * (i - left) / 2;
                }
            }
            s.push(i);
        }
        while (!s.empty()) {
            int cur = s.top();
            s.pop();
            int left = s.empty() ? -1 : s.top();
            int bottom = s.empty() ? 0 : heights[left];
            ans += (heights[cur] - bottom) * (heights.size() - left - 1) * (heights.size() - left) / 2;
        }
        return ans;
    }
};
```
