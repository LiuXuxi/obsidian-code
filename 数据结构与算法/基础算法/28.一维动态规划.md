# Kadane算法

模板

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

/**
 * @brief Kadane 算法通用模板
 * @param vec 转化后的增益数组
 * @return 最大连续子段和 (如果允许空段则初始 ans=0, 否则初始 ans=极小值)
 */
ll kadane(const vector<ll>& vec) {
    ll max_so_far = 0;  // 全局最大 (如果必须选一个数，设为 vec[0] 或 LLONG_MIN)
    ll current_max = 0; // 以当前位置结尾的最大和

    for (ll x : vec) {
        current_max += x;
        if (max_so_far < current_max) {
            max_so_far = current_max;
        }
        if (current_max < 0) {
            current_max = 0; // 核心：如果变负了，不如从 0 开始
        }
    }
    return max_so_far;
}

int main() {
    // 逻辑演示：区间替换问题
    int n; ll x_val;
    cin >> n >> x_val;
    
    vector<ll> gains;
    ll original_sum = 0;
    
    for(int i = 0; i < n; ++i) {
        ll a; cin >> a;
        original_sum += a;
        gains.push_back(x_val - a); // 转化为增益序列
    }
    
    cout << original_sum + kadane(gains) << endl;
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,dp[10],maxn;
string a;//为了方便取头尾，可以以字符串形式存储
signed main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a;
		int ln=a.length();
		dp[a[ln-1]-'0']=max(dp[a[ln-1]-'0'],dp[a[0]-'0']+1);
	}
	for(int i=0;i<=9;i++) maxn=max(maxn,dp[i]);
	cout<<n-maxn;
	return 0;
}
```

[U549625 小苯的区间和疑惑 - 洛谷](https://www.luogu.com.cn/problem/U549625#ide)
```cpp
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

long long arr[200005];
long long L[200005]; // 以 i 结尾的最大子段和
long long R[200005]; // 以 i 开始的最大子段和

void solve() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> arr[i];

    // 从左往右跑 Kadane
    // L[i] = max(arr[i], L[i-1] + arr[i])
    L[0] = -2e18; // 极小值初始化
    long long current_max = 0;
    for (int i = 1; i <= n; i++) {
        current_max = max(arr[i], current_max + arr[i]);
        L[i] = current_max;
    }

    // 从右往左跑 Kadane
    // R[i] = max(arr[i], R[i+1] + arr[i])
    R[n + 1] = -2e18;
    current_max = 0;
    for (int i = n; i >= 1; i--) {
        current_max = max(arr[i], current_max + arr[i]);
        R[i] = current_max;
    }

    // 输出经过 i 的最大和
    for (int i = 1; i <= n; i++) {
        cout << (L[i] + R[i] - arr[i]) << (i == n ? "" : " ");
    }
    cout << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    return 0;
}
```

[P1115 最大子段和 - 洛谷](https://www.luogu.com.cn/problem/solution/P1115)
```cpp
#include<iostream>

#include<limits.h>

using namespace std;

  

using ll=long long;

  

ll cur;

ll ans=INT_MIN;

int n,a;

  

int main(){

    ios::sync_with_stdio(false);

    cin.tie(0);

    cin>>n;

    for(int i=1;i<=n;i++){

        cin>>a;

        cur+=a;

        ans=max(ans,cur);

        cur=max(cur,0LL);

    }

    cout<<ans<<endl;

    return 0;

}
```
