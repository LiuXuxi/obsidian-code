[3412. 计算字符串的镜像分数 - 力扣（LeetCode）](https://leetcode.cn/problems/find-mirror-score-of-a-string/submissions/720713697/)
>维护所有含有相同Key值的Val，但查找一个符合要求的val：
> 先进先出：stack
> 先进后出：queue
> 需要维护多个结构，可以用哈希表也可以用数组

哈希表维护：
```cpp
class Solution {

public:

    char mirror(char&c){

        return (25-(c-'a'))+'a';

    }

  

    long long calculateScore(string s) {

        unordered_map<char,stack<int>>cnt;

        int n=s.size();

        long long ans=0;

        for(int i=0;i<n;i++){

            char val=mirror(s[i]);

            if(cnt.contains(val)){

                ans+=i-cnt[val].top();

                cnt.erase(val);

            }else cnt[s[i]].push(i);

        }

        return ans;

    }

};
```
数组维护：
```cpp
class Solution {
public:
    long long calculateScore(string s) {
        stack<int> stk[26];
        long long ans = 0;
        for (int i = 0; i < s.size(); i++) {
            int c = s[i] - 'a';
            if (!stk[25 - c].empty()) {
                ans += i - stk[25 - c].top();
                stk[25 - c].pop();
            } else {
                stk[c].push(i);                
            }
        }
        return ans;
    }
};
```

[1172. 餐盘栈 - 力扣（LeetCode）](https://leetcode.cn/problems/dinner-plate-stacks/solutions/2248707/yu-qi-wei-hu-di-yi-ge-wei-man-zhan-bu-ru-sphs/)
> 维护所有未满的栈
```cpp
class DinnerPlates {

    int capacity; // 栈的容量

    vector<stack<int>> stacks; // 所有栈

    priority_queue<int, vector<int>, greater<>> idx; // 未满栈的下标

public:

    DinnerPlates(int capacity) : capacity(capacity) {}

  

    void push(int val) {

        if (!idx.empty() && idx.top() >= stacks.size())

            while (!idx.empty()) idx.pop(); // 堆中都是越界下标，直接清空

        if (idx.empty()) { // 所有栈都是满的

            stack<int> st;

            st.push(val);

            stacks.emplace_back(st); // 添加一个新的栈

            if (capacity > 1) // 新的栈没有满

                idx.push(stacks.size() - 1); // 入堆

        } else { // 还有未满栈

            auto &st = stacks[idx.top()];

            st.push(val); // 入栈

            if (st.size() == capacity) // 栈满了

                idx.pop(); // 从堆中去掉

        }

    }

  

    int pop() {

        // 等价为 popAtStack 最后一个非空栈

        return popAtStack(stacks.size() - 1);

    }

  

    int popAtStack(int index) {

        if (index < 0 || index >= stacks.size() || stacks[index].empty())

            return -1; // 非法操作

        auto &st = stacks[index];

        if (st.size() == capacity) // 满栈

            idx.push(index); // 元素出栈后，栈就不满了，把下标入堆

        int val = st.top();

        st.pop();

        while (!stacks.empty() && stacks.back().empty())

            stacks.pop_back(); // 去掉末尾的空栈（懒删除，堆中下标在 push 时处理）

        return val;

    }

};
```
