[3412. 计算字符串的镜像分数 - 力扣（LeetCode）](https://leetcode.cn/problems/find-mirror-score-of-a-string/submissions/720713697/)
>维护所有含有相同Key值的Val，但查找一个符合要求的val：
> 先进先出：stack
> 先进后出：queue
> 需要维护多个结构，可以用哈希表也可以用数组

哈希表维护：
```cpp
class Solution {

public:

    char mirror(char&c){

        return (25-(c-'a'))+'a';

    }

  

    long long calculateScore(string s) {

        unordered_map<char,stack<int>>cnt;

        int n=s.size();

        long long ans=0;

        for(int i=0;i<n;i++){

            char val=mirror(s[i]);

            if(cnt.contains(val)){

                ans+=i-cnt[val].top();

                cnt.erase(val);

            }else cnt[s[i]].push(i);

        }

        return ans;

    }

};
```
数组维护：
```cpp
class Solution {
public:
    long long calculateScore(string s) {
        stack<int> stk[26];
        long long ans = 0;
        for (int i = 0; i < s.size(); i++) {
            int c = s[i] - 'a';
            if (!stk[25 - c].empty()) {
                ans += i - stk[25 - c].top();
                stk[25 - c].pop();
            } else {
                stk[c].push(i);                
            }
        }
        return ans;
    }
};
```

[1172. 餐盘栈 - 力扣（LeetCode）](https://leetcode.cn/problems/dinner-plate-stacks/solutions/2248707/yu-qi-wei-hu-di-yi-ge-wei-man-zhan-bu-ru-sphs/)
> 维护所有未满的栈
```cpp
class DinnerPlates {

    int capacity; // 栈的容量

    vector<stack<int>> stacks; // 所有栈

    priority_queue<int, vector<int>, greater<>> idx; // 未满栈的下标

public:

    DinnerPlates(int capacity) : capacity(capacity) {}

  

    void push(int val) {

        if (!idx.empty() && idx.top() >= stacks.size())

            while (!idx.empty()) idx.pop(); // 堆中都是越界下标，直接清空

        if (idx.empty()) { // 所有栈都是满的

            stack<int> st;

            st.push(val);

            stacks.emplace_back(st); // 添加一个新的栈

            if (capacity > 1) // 新的栈没有满

                idx.push(stacks.size() - 1); // 入堆

        } else { // 还有未满栈

            auto &st = stacks[idx.top()];

            st.push(val); // 入栈

            if (st.size() == capacity) // 栈满了

                idx.pop(); // 从堆中去掉

        }

    }

  

    int pop() {

        // 等价为 popAtStack 最后一个非空栈

        return popAtStack(stacks.size() - 1);

    }

  

    int popAtStack(int index) {

        if (index < 0 || index >= stacks.size() || stacks[index].empty())

            return -1; // 非法操作

        auto &st = stacks[index];

        if (st.size() == capacity) // 满栈

            idx.push(index); // 元素出栈后，栈就不满了，把下标入堆

        int val = st.top();

        st.pop();

        while (!stacks.empty() && stacks.back().empty())

            stacks.pop_back(); // 去掉末尾的空栈（懒删除，堆中下标在 push 时处理）

        return val;

    }

};
```

[2589. 完成所有任务的最少时间 - 力扣（LeetCode）](https://leetcode.cn/problems/minimum-time-to-complete-all-tasks/)
> 贪心+暴力

```cpp
class Solution {

public:

    int findMinimumTime(vector<vector<int>>& tasks) {

        ranges::sort(tasks, {}, [](auto& a) { return a[1]; }); // 按照右端点从小到大排序

        int ans = 0;

        vector<int> run(tasks.back()[1] + 1);

        for (auto& t : tasks) {

            int start = t[0], end = t[1], d = t[2];

            d -= reduce(run.begin() + start, run.begin() + end + 1); // 去掉运行中的时间点

            for (int i = end; d > 0; i--) { // 剩余的 d 填充区间后缀

                if (!run[i]) {

                    run[i] = true; // 运行

                    d--;

                    ans++;

                }

            }

        }

        return ans;

    }

};
```

> 栈+二分+前缀和
> 栈维护区间以及前缀和
> 二分查找在栈中查找符合条件的区间

```cpp
class Solution {

public:

    int findMinimumTime(vector<vector<int>>& tasks) {

        // 1. 贪心前提：按照任务的右端点（结束时间）从小到大排序。

        // 只有先处理早结束的任务，才能让它们的时间点尽量靠后，从而被后面晚开始的任务复用。

        ranges::sort(tasks, {}, [](auto& a) { return a[1]; });

  

        // 2. st 是一个单调栈，存储的是一系列【互不相交】且【从左到右排列】的运行时间段。

        // array<int, 3> 的含义：

        // [0]: l (区间左端点)

        // [1]: r (区间右端点)

        // [2]: s (前缀和：从栈底到当前区间为止，所有区间包含的总时间点数量)

        // 初始哨兵 {-2, -2, 0} 是为了让 lower_bound 永远能找到有效位置，且不影响计算。

        vector<array<int, 3>> st = {{-2, -2, 0}};

  

        for (auto& t : tasks) {

            int start = t[0]; // 当前任务开始时间

            int end = t[1];   // 当前任务结束时间

            int d = t[2];     // 当前任务还需要运行的时间天数

  

            // 3. 计算 [start, end] 范围内已经有多少天在运行了。

            // 使用二分查找：在栈中找第一个左端点 l >= start 的区间位置。

            // 然后通过 -- 操作移到前一个位置，此时这个区间是最后一个【左端点 < start】的区间。

            auto it = --ranges::lower_bound(st, start, {}, [](auto& x) { return x[0]; });

            auto [_, r, s] = *it;

  

            // 通过前缀和快速计算：总已用时间 - 查找到的区间之前积累的时间 = [start之后]的已用时间

            d -= st.back()[2] - s;

  

            // 特殊情况：如果 start 落在查找到的那个区间的内部 (start <= r)

            // 说明那个区间里有一部分点（从 start 到 r）也是可以被当前任务复用的。

            if (start <= r) {

                d -= r - start + 1;

            }

  

            // 4. 如果 d <= 0，说明当前任务在之前任务运行的过程中已经顺便完成了，直接跳过。

            if (d <= 0) {

                continue;

            }

  

            // 5. 重点：贪心填充。

            // 我们需要从当前任务的 end 开始【逆向】填充 d 个点。

            // 如果新填的区间会和栈顶的旧区间发生“碰撞”或“重叠”，就合并它们。

            // end - st.back()[1] <= d 的意思是：

            // 当前任务终点与栈顶区间终点的距离，小于等于我还要填入的天数，

            // 这意味着新填的区域必然会覆盖或连接栈顶区间。

            while (end - st.back()[1] <= d) {

                auto [l, r, _] = st.back();

                st.pop_back();

                // 合并旧区间：把旧区间的长度加回 d，相当于我要从 end 开始重新画一个更大的连续区间

                d += r - l + 1;

            }

  

            // 6. 将新产生的一整段运行区间压入栈中。

            // 新区间的左端点就是 end 往前数 d 个点，右端点是 end。

            // 前缀和更新为：栈顶前缀和 + 本次新增的 d。

            st.push_back({end - d + 1, end, st.back()[2] + d});

        }

  

        // 栈顶的最后一个前缀和就是所有任务运行的总天数。

        return st.back()[2];

    }

};
```

[1003. 检查替换后的词是否有效 - 力扣（LeetCode）](https://leetcode.cn/problems/check-if-word-is-valid-after-substitutions/solutions/2253773/zhan-jian-ji-xie-fa-pythonjavacgo-by-end-i9o7/)
> 考虑多邻项的消除
> 简化：找相同的消除逻辑
```cpp
class Solution {

public:

    bool isValid(string s) {

        int i=0;

        for(auto&c:s){

            if(c>'a'&&(i==0||c-s[--i]!=1))return false;

            if(c<'c')s[i++]=c;

        }

        return i==0;

    }

};
```

[735. 小行星碰撞 - 力扣（LeetCode）](https://leetcode.cn/problems/asteroid-collision/)
> 分多种类讨论邻项消除
```cpp
class Solution {

public:

    vector<int> asteroidCollision(vector<int>& asteroids) {

        vector<int>ans;

        for(auto&cur:asteroids){

            if(cur>0)ans.push_back(cur);

            else{

                bool flag=false;

                while(!ans.empty()&&ans.back()>0){

                    int top=ans.back();

                    if(top<=-cur)ans.pop_back();

                    if(top>=-cur){

                        flag=true;

                        break;

                    }

                }

                if(!flag)ans.push_back(cur);

            }

        }

        return ans;

    }

};
```
